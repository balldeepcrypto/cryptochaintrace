  <h1>CryptoChainTrace</h1>

  <select id="chain">

    <!-- We can add other chains later -->
  </select>

  <input type="text" id="address" placeholder="Paste DAG wallet address here" style="width:80%">

  <button onclick="loadWallet()">LOAD WALLET</button>

  <div id="status" class="status"></div>
  <div id="result" class="result"></div>

  <script>
    let allTxs = [];
    let nextCursor = null;

    async function loadWallet(isMore = false) {
      const address = document.getElementById("address").value.trim();
      const status = document.getElementById("status");
      const result = document.getElementById("result");

      if (!address) {
        status.innerHTML = "Please enter a wallet address";
        return;
      }

      status.innerHTML = isMore ? "Loading more (+100)..." : "Fetching transactions...";

      try {
        let url = `https://be-mainnet.constellationnetwork.io/addresses/${address}/transactions?limit=100`;
        if (nextCursor) url += `&search_after=${nextCursor}`;

        const res = await fetch(url);
        if (!res.ok) throw new Error("API error");

        const data = await res.json();
        const newTxs = data.data || data.transactions || [];

        allTxs = allTxs.concat(newTxs);   // Append only
        nextCursor = data.meta?.next || data.nextCursor || null;

        let html = `<h2>DAG Wallet: ${address}</h2>`;
        html += `<p class="status">${allTxs.length} transactions loaded ${nextCursor ? '• More available' : '• Full History'}</p>`;

        allTxs.forEach(tx => {
          const dir = (tx.source === address || tx.from === address) ? "OUT" : "IN";
          html += `
            <div class="tx">
              <strong>${dir}</strong> — ${tx.amount || '—'} DAG<br>
              Counterparty: ${tx.destination || tx.to || tx.source || '—'}<br>
              Date: ${tx.timestamp ? new Date(tx.timestamp).toLocaleString() : '—'}
            </div>`;
        });

        if (nextCursor) {
          html += `<button onclick="loadWallet(true)" style="margin-top:20px; padding:15px;">LOAD MORE (+100)</button>`;
        }

        result.innerHTML = html;
        status.innerHTML = "";

      } catch (err) {
        status.innerHTML = `Error: ${err.message}. Try again.`;
      }
    }
  </script>
