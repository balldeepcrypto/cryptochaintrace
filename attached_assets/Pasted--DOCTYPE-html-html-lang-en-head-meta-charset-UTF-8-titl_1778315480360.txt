  <button onclick="loadWallet()">LOAD WALLET</button>

  <div id="status"></div>
  <div id="result" class="result"></div>

  <script>
    let allTxs = [];
    let nextCursor = null;

    async function loadWallet(more = false) {
      const address = document.getElementById("address").value.trim();
      const status = document.getElementById("status");
      const result = document.getElementById("result");

      if (!address) return;

      status.innerHTML = more ? "Loading more..." : "Fetching transactions...";

      try {
        let url = `https://be-mainnet.constellationnetwork.io/addresses/${address}/transactions?limit=300`;
        if (nextCursor) url += `&search_after=${nextCursor}`;

        const res = await fetch(url);
        const data = await res.json();

        const newTxs = data.transactions || [];
        allTxs = allTxs.concat(newTxs);   // <-- Append only

        nextCursor = data.nextCursor || null;

        let html = `<h2>DAG Wallet: ${address}</h2><p>${allTxs.length} transactions loaded</p>`;

        allTxs.forEach(tx => {
          const dir = tx.source === address ? "OUT" : "IN";
          html += `
            <div class="tx">
              <strong>${dir}</strong> — ${tx.amount || '—'} DAG<br>
              Counterparty: ${tx.destination || tx.source || '—'}<br>
              Date: ${tx.timestamp ? new Date(tx.timestamp).toLocaleString() : '—'}
            </div>`;
        });

        if (nextCursor) {
          html += `<button onclick="loadWallet(true)">LOAD MORE (+300)</button>`;
        } else {
          html += `<p><strong>Full history loaded.</strong></p>`;
        }

        result.innerHTML = html;
        status.innerHTML = "";

      } catch (err) {
        status.innerHTML = `Error: ${err.message}. Try again.`;
      }
    }
  </script>
